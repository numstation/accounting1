        // Suggest account code based on selected account type

        function suggestAccountCode(accountType) {

            const accountCodeInput = document.getElementById('account-code');

            if (!accountCodeInput || accountCodeInput.value) return; // Don't suggest if already filled

            

            let codePrefix = '';

            let nextNumber = 1;

            

            switch (accountType) {

                case 'asset':

                    codePrefix = '1';

                    // Find next available number

                    if (chartOfAccounts.asset.length > 0) {

                        const highestCode = Math.max(...chartOfAccounts.asset.map(a => parseInt(a.code)));

                        nextNumber = (highestCode % 100) + 1;

                    }

                    break;

                case 'liability':

                    codePrefix = '2';

                    if (chartOfAccounts.liability.length > 0) {

                        const highestCode = Math.max(...chartOfAccounts.liability.map(a => parseInt(a.code)));

                        nextNumber = (highestCode % 100) + 1;

                    }

                    break;

                case 'equity':

                    codePrefix = '3';

                    if (chartOfAccounts.equity.length > 0) {

                        const highestCode = Math.max(...chartOfAccounts.equity.map(a => parseInt(a.code)));

                        nextNumber = (highestCode % 100) + 1;

                    }

                    break;

                case 'revenue':

                    codePrefix = '4';

                    if (chartOfAccounts.revenue.length > 0) {

                        const highestCode = Math.max(...chartOfAccounts.revenue.map(a => parseInt(a.code)));

                        nextNumber = (highestCode % 100) + 1;

                    }

                    break;

                case 'expense':

                    codePrefix = '5';

                    if (chartOfAccounts.expense.length > 0) {

                        const highestCode = Math.max(...chartOfAccounts.expense.map(a => parseInt(a.code)));

                        nextNumber = (highestCode % 100) + 1;

                    }

                    break;

            }

            

            if (codePrefix) {

                // Format as 3-digit code (e.g., 101, 201, etc.)

                accountCodeInput.value = codePrefix + String(nextNumber).padStart(2, '0');

            }

        }

        function getDateRangeForReport() {

            const dateRange = document.getElementById('report-date-range').value;

            const today = new Date();

            let startDate, endDate;

            

            switch (dateRange) {

                case 'current-month':

                    startDate = new Date(today.getFullYear(), today.getMonth(), 1);

                    endDate = new Date(today.getFullYear(), today.getMonth() + 1, 0);

                    break;

                case 'current-quarter':

                    const quarter = Math.floor(today.getMonth() / 3);

                    startDate = new Date(today.getFullYear(), quarter * 3, 1);

                    endDate = new Date(today.getFullYear(), (quarter + 1) * 3, 0);

                    break;

                case 'current-year':

                    startDate = new Date(today.getFullYear(), 0, 1);

                    endDate = new Date(today.getFullYear(), 11, 31);

                    break;

                case 'last-month':

                    startDate = new Date(today.getFullYear(), today.getMonth() - 1, 1);

                    endDate = new Date(today.getFullYear(), today.getMonth(), 0);

                    break;

                case 'last-quarter':

                    const lastQuarter = Math.floor(today.getMonth() / 3) - 1;

                    const yearOffset = lastQuarter < 0 ? -1 : 0;

                    const adjustedQuarter = lastQuarter < 0 ? 3 + lastQuarter : lastQuarter;

                    startDate = new Date(today.getFullYear() + yearOffset, adjustedQuarter * 3, 1);

                    endDate = new Date(today.getFullYear() + yearOffset, (adjustedQuarter + 1) * 3, 0);

                    break;

                case 'last-year':

                    startDate = new Date(today.getFullYear() - 1, 0, 1);

                    endDate = new Date(today.getFullYear() - 1, 11, 31);

                    break;

                case 'custom':

                    startDate = new Date(document.getElementById('custom-start-date').value);

                    endDate = new Date(document.getElementById('custom-end-date').value);

                    break;

                default:

                    startDate = new Date(today.getFullYear(), 0, 1);

                    endDate = today;

            }

            

            return { startDate, endDate };

        }

        

        function updateReport() {

            if (!currentReport) return;

            

            const reportContent = document.getElementById('report-content');

            if (!reportContent) return;

            

            const { startDate, endDate } = getDateRangeForReport();

            const reportFormat = document.getElementById('report-format').value;

            

            let reportHtml = '';

            

            switch (currentReport) {

                case 'income-statement':

                    reportHtml = generateIncomeStatementReport(startDate, endDate, reportFormat);

                    break;

                case 'balance-sheet':

                    reportHtml = generateBalanceSheetReport(endDate, reportFormat);

                    break;

                case 'cash-flow':

                    reportHtml = generateCashFlowReport(startDate, endDate, reportFormat);

                    break;

                case 'general-ledger':

                    const accountId = document.getElementById('report-account').value;

                    reportHtml = generateGeneralLedgerReport(startDate, endDate, accountId, reportFormat);

                    break;

                case 'trial-balance':

                    reportHtml = generateTrialBalanceReport(endDate, reportFormat);

                    break;

                case 'tax-summary':

                    reportHtml = generateTaxSummaryReport(startDate, endDate, reportFormat);

                    break;

                default:

                    reportHtml = '<p>Report not implemented yet.</p>';

            }

            

            reportContent.innerHTML = reportHtml;

        }
